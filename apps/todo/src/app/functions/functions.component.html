Functions and objects

Functions

// Named function
function greet(name: string) {
 return 'hello ' + name
}

// Function expression
let greet2 = function(name: string) {
 return 'hello ' + name
}

// Arrow function expression
let greet3 = (name: string) => {
  return 'hello ' + name
}
